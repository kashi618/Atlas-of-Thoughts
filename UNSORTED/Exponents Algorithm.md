```
int x     (number)
int n     (exponent)
int i = 1
int ans

// Finds the answer of x^n
power(x,n)
	// if the exponent is 0, then the answer is always 1
	if n = 0
		ans = 1
	
	// multiplies the number x, n amount of times
	else
		while i < n
			ans = ans * x
			i++
		END WHILE
	END ELSE
	
	return ans
```
