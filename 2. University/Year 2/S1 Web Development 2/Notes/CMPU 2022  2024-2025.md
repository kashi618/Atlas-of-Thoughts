```php
<?php
$conn = new mysqli("localhost", "root", "", "AerAirways");
if ($conn->connect_error) {
	die("Connection failed: ".$conn->connect_error);
}

$origin = $_GET['going_from'];

$sql = "SELECT Airport.country, Flight.flightNum
		FROM Flight JOIN Airport
		ON Flight.goingTo = Airport.airport
		WHERE Flight.GOingFrom = '$origin'
		ORDER BY Airport.country, Flight.flightNum;";
$result = mysqli_query($conn, $sql);

$destination = [];
while ($row = $result->fetch_assoc()) {
	$destination[$row['country']][] = $row['flightNum'];
}

echo "<ul>";
foreach ($destination as $country => $flights) {
	echo "<li>" . $country . "<ul>";
	foreach ($flights as $flight) {
		echo "<li>" . $flight . "</li>";
	}
	echo "</ul></li>";
}
echo "</ul>";

mysqli_close($conn);
?>
```
